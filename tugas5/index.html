mixin CanFly {
  bool _isFlying = false; // private

  bool get isFlying => _isFlying;

  void fly() {
    if (!_isFlying) {
      _isFlying = true;
      print('Started flying!');
    } else {
      print('Already flying!');
    }
  }

  void land() {
    if (_isFlying) {
      _isFlying = false;
      print('Landed!');
    } else {
      print('Already on the ground!');
    }
  }
}

mixin CanSwim {
  bool _isSwimming = false; // private

  bool get isSwimming => _isSwimming;

  void swim() {
    if (!_isSwimming) {
      _isSwimming = true;
      print('Started swimming!');
    } else {
      print('Already swimming!');
    }
  }

  void stopSwim() {
    if (_isSwimming) {
      _isSwimming = false;
      print('Stop swimming!');
    } else {
      print('Already not swimming!');
    }
  }
}

import 'dart:math';

class MathUtils {
  // 1. Faktorial
  static int factorial(int n) {
    if (n < 0) {
      throw ArgumentError("Input tidak boleh negatif");
    }
    int result = 1;
    for (int i = 1; i <= n; i++) {
      result *= i;
    }
    return result;
  }

  // 2. Cek bilangan prima
  static bool isPrime(int n) {
    if (n <= 1) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;

    for (int i = 3; i <= sqrt(n).toInt(); i += 2) {
      if (n % i == 0) return false;
    }
    return true;
  }

  // 3. Membulatkan double ke desimal tertentu
  static double roundTo(double value, int decimals) {
    double factor = pow(10, decimals).toDouble();
    return (value * factor).round() / factor;
  }

  // 4. Random integer dalam range [min, max]
  static int randomInt(int min, int max) {
    Random rand = Random();
    return min + rand.nextInt(max - min + 1);
  }

  // 5. Random double dalam range [min, max]
  static double randomDouble(double min, double max) {
    Random rand = Random();
    return min + rand.nextDouble() * (max - min);
  }

  // 6. Rata-rata (mean)
  static double mean(List numbers) {
    if (numbers.isEmpty) {
      throw ArgumentError("List tidak boleh kosong");
    }
    double sum = numbers.reduce((a, b) => a + b);
    return sum / numbers.length;
  }

  // 7. Median
  static double median(List numbers) {
    if (numbers.isEmpty) {
      throw ArgumentError("List tidak boleh kosong");
    }

    List sorted = List.from(numbers)..sort();
    int n = sorted.length;

    if (n % 2 == 1) {
      return sorted[n ~/ 2];
    } else {
      return (sorted[n ~/ 2 - 1] + sorted[n ~/ 2]) / 2;
    }
  }
}

// Contoh penggunaan
void main() {
  print("Faktorial 5: ${MathUtils.factorial(5)}");
  print("Apakah 7 prima? ${MathUtils.isPrime(7)}");
  print("Pembulatan 3.14159 ke 2 desimal: ${MathUtils.roundTo(3.14159, 2)}");
  print("Random int (1-10): ${MathUtils.randomInt(1, 10)}");
  print("Random double (0-1): ${MathUtils.randomDouble(0, 1)}");

  List data = [1, 2, 3, 4, 5];
  print("Mean: ${MathUtils.mean(data)}");
  print("Median: ${MathUtils.median(data)}");
}

  }

// Abstract class MediaPlayer
abstract class MediaPlayer {
  String name;
  int duration; // in seconds
  
  MediaPlayer(this.name, this.duration);
  
  void displayInfo();
}

// Interface Playable (using abstract class in Dart)
abstract class Playable {
  void play();
  void stop();
}

// Interface Pausable
abstract class Pausable {
  void pause();
  void resume();
}

// AudioPlayer implementation
class AudioPlayer extends MediaPlayer implements Playable, Pausable {
  bool _isPlaying = false;
  bool _isPaused = false;
  int _currentPosition = 0;
  
  AudioPlayer(String name, int duration) : super(name, duration);
  
  @override
  void play() {
    if (_isPaused) {
      print('Resuming audio: $name');
      _isPaused = false;
      _isPlaying = true;
    } else if (!_isPlaying) {
      print('Playing audio: $name (Duration: ${duration} seconds)');
      _isPlaying = true;
      _currentPosition = 0;
    } else {
      print('Audio $name is already playing');
    }
  }
  
  @override
  void stop() {
    if (_isPlaying || _isPaused) {
      print('Stopping audio: $name');
      _isPlaying = false;
      _isPaused = false;
      _currentPosition = 0;
    } else {
      print('Audio $name is not playing');
    }
  }
  
  @override
  void pause() {
    if (_isPlaying && !_isPaused) {
      print('Pausing audio: $name at position $_currentPosition seconds');
      _isPlaying = false;
      _isPaused = true;
    } else if (!_isPlaying) {
      print('Cannot pause $name - audio is not playing');
    } else {
      print('Audio $name is already paused');
    }
  }
  
  @override
  void resume() {
    if (_isPaused) {
      print('Resuming audio: $name from position $_currentPosition seconds');
      _isPlaying = true;
      _isPaused = false;
    } else {
      print('Cannot resume $name - audio is not paused');
    }
  }
  
  @override
  void displayInfo() {
    print('=== Audio Player Info ===');
    print('Name: $name');
    print('Duration: ${duration} seconds');
    print('Status: ${_isPlaying ? 'Playing' : (_isPaused ? 'Paused' : 'Stopped')}');
    print('Current position: $_currentPosition seconds');
    print('========================\n');
  }
}

// VideoPlayer implementation
class VideoPlayer extends MediaPlayer implements Playable, Pausable {
  bool _isPlaying = false;
  bool _isPaused = false;
  int _currentPosition = 0;
  String _quality = '720p';
  
  VideoPlayer(String name, int duration) : super(name, duration);
  
  void setQuality(String quality) {
    _quality = quality;
    print('Video quality set to: $_quality');
  }
  
  @override
  void play() {
    if (_isPaused) {
      print('Resuming video: $name at $_quality quality');
      _isPaused = false;
      _isPlaying = true;
    } else if (!_isPlaying) {
      print('Playing video: $name (Duration: ${duration} seconds, Quality: $_quality)');
      _isPlaying = true;
      _currentPosition = 0;
    } else {
      print('Video $name is already playing');
    }
  }
  
  @override
  void stop() {
    if (_isPlaying || _isPaused) {
      print('Stopping video: $name');
      _isPlaying = false;
      _isPaused = false;
      _currentPosition = 0;
    } else {
      print('Video $name is not playing');
    }
  }
  
  @override
  void pause() {
    if (_isPlaying && !_isPaused) {
      print('Pausing video: $name at position $_currentPosition seconds');
      _isPlaying = false;
      _isPaused = true;
    } else if (!_isPlaying) {
      print('Cannot pause $name - video is not playing');
    } else {
      print('Video $name is already paused');
    }
  }
  
  @override
  void resume() {
    if (_isPaused) {
      print('Resuming video: $name from position $_currentPosition seconds');
      _isPlaying = true;
      _isPaused = false;
    } else {
      print('Cannot resume $name - video is not paused');
    }
  }
  
  @override
  void displayInfo() {
    print('=== Video Player Info ===');
    print('Name: $name');
    print('Duration: ${duration} seconds');
    print('Quality: $_quality');
    print('Status: ${_isPlaying ? 'Playing' : (_isPaused ? 'Paused' : 'Stopped')}');
    print('Current position: $_currentPosition seconds');
    print('========================\n');
  }
}

// Main function to demonstrate the system
void main() {
  print('=== MEDIA PLAYER SYSTEM DEMO ===\n');
  
  // Create instances
  var audioPlayer = AudioPlayer('Summer Vibes.mp3', 180);
  var videoPlayer = VideoPlayer('Nature Documentary.mp4', 360);
  
  // Display initial info
  audioPlayer.displayInfo();
  videoPlayer.displayInfo();
  
  // Test AudioPlayer
  print('--- Testing AudioPlayer ---');
  audioPlayer.play();
  audioPlayer.pause();
  audioPlayer.resume();
  audioPlayer.stop();
  audioPlayer.displayInfo();
  
  print('\n--- Testing VideoPlayer ---');
  videoPlayer.play();
  videoPlayer.setQuality('1080p');
  videoPlayer.pause();
  videoPlayer.resume();
  videoPlayer.stop();
  videoPlayer.displayInfo();
  
  print('\n--- Polymorphism Test ---');
  List players = [audioPlayer, videoPlayer];
  
  for (var player in players) {
    player.displayInfo();
    if (player is Playable) {
      print('This player supports Playable interface');
      (player as Playable).play();
      (player as Playable).stop();
    }
    if (player is Pausable) {
      print('This player supports Pausable interface');
    }
    print('---');
  }
}

Dengan menerapkan kedua konsep ini, kode menjadi lebih mudah dikembangkan, digunakan kembali, serta lebih efisien dalam pengelolaannya.
