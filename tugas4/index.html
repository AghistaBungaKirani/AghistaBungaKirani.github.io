Review Sesi 1: OOP adalah paradigma berfokus pada objek. Kelas = blueprint, objek = instance. Constructor digunakan untuk inisialisasi objek.

Jenis Constructor di Dart:

  • Default Constructor: ClassName()
  • Named Constructor: ClassName.name()
  • Parameterized Constructor: ClassName(this.param)
  • Constant Constructor: const ClassName() (semua field final, immutable, hemat memori)
  • Factory Constructor: (lanjutan)

Enkapsulasi: Menyembunyikan detail implementasi dengan underscore _. Tujuan: keamanan, kontrol akses, fleksibilitas.

Getter & Setter: Memberikan akses terkontrol ke field private, bisa ditambahi validasi.

Named Constructor-Contoh Mobil

Named Constructor adalah konstruktor yang memiliki nama tambahan selain nama kelas. Named constructor digunakan jika kita ingin membuat objek dengan cara yang berbeda dalam satu kelas.

 //Nama: Aghista Bunga Kirani
//Nim: 251410027
//Kelas: SI2A


import 'dart:convert';
class Mobil {
  String? merk;
  int? tahun;

  // Parameterized constructor
  Mobil(this.merk, this.tahun);

  // Named constructor tanpa parameter
  Mobil.baru() {
    merk = "Toyota";
    tahun = 2023;
  }

  // Named constructor dengan parameter JSON
  Mobil.dariJson(Map json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }

  @override
  String toString() {
    return "Mobil(merk: $merk, tahun: $tahun)";
  }
}

void main() {
  // Membuat objek dengan constructor biasa
  var mobil1 = Mobil("Honda", 2022);

  // Membuat objek dengan named constructor
  var mobil2 = Mobil.baru();

  // Membuat objek dari JSON
  var dataJson = '{"merk":"Suzuki","tahun":2020}';
  var mobil3 = Mobil.dariJson(jsonDecode(dataJson));

  print(mobil1);
  print(mobil2);
  print(mobil3);
}

constant contructor-Warna

Constant Constructor adalah konstruktor dalam Dart yang digunakan untuk membuat objek yang bersifat immutable (tidak dapat diubah setelah dibuat). Konstruktor ini menggunakan kata kunci const, dan semua atribut dalam class harus bertipe final.

 //Nama: Aghista Bunga Kirani
//Nim: 251410027
//Kelas: SI2A


class Warna {
  final int r;
  final int g;
  final int b;

  // Constant Constructor
  const Warna(this.r, this.g, this.b);

  @override
  String toString() {
    return "Warna(r: $r, g: $g, b: $b)";
  }
}

void main() {
  // Membuat objek warna menggunakan const
  const merah = Warna(255, 0, 0);
  const merahLagi = Warna(255, 0, 0);

  print(merah);
  print(merahLagi);

  // Membuktikan bahwa kedua objek menggunakan referensi yang sama
  print(identical(merah, merahLagi));
}

}

Enkapsulasi

Konsep: Enkapsulasi adalah konsep dalam pemrograman berorientasi objek yang digunakan untuk menyembunyikan detail implementasi dari luar kelas. Data di dalam kelas tidak bisa diakses secara langsung, tetapi melalui metode tertentu.

Tujuan: Tujuan: Keamanan data – Melindungi data agar tidak diubah sembarangan. Kontrol akses – Mengatur bagaimana data dapat diakses atau diubah. Fleksibilitas perubahan – Memudahkan perubahan kode tanpa mempengaruhi bagian lain dari program.

Cara di Dart: Menggunakan underscore (_) di depan nama variabel atau metode untuk menjadikannya private members.

Publik VS Private

Public adalah variabel atau metode yang dapat diakses dari mana saja, baik dari dalam kelas maupun dari luar kelas atau file lain. Private adalah variabel atau metode yang hanya dapat diakses di dalam file yang sama. Di Dart, private dibuat dengan menambahkan underscore (_) di depan nama variabel atau metode.

//Nama: Aghista Bunga Kirani
//Nim: 251410027
//Kelas: SI2A


class Person {
  String name;   // public
  int _age;      // private

  // constructor
  Person(this.name, this._age);

  void tampilkanData() {
    print("Nama: $name");
    print("Umur: $_age");
  }
}

void main() {
  Person p = Person("Aghista", 19);

  // bisa diakses
  print(p.name);

  // tidak bisa diakses langsung karena private
  // print(p._age); // akan error

  p.tampilkanData();
}

Contoh Setter & Getter

Getter adalah metode yang digunakan untuk mengambil atau membaca nilai dari variabel private di dalam sebuah class.

Setter adalah metode yang digunakan untuk mengubah atau memberi nilai pada variabel private di dalam class.

//Nama: Aghista Bunga Kirani
//Nim: 251410027
//Kelas: SI2A


class Produk {
  String _nama;
  double _harga;
  
  Produk(this._nama, this._harga);
  
  // Getter
  String get nama => _nama;
  double get harga => _harga;
  
  // Setter dengan validasi
  set nama(String value) {
    if (value.length >= 3) {
      _nama = value;
    } else {
      print('Gagal memperbarui nama: Nama harus memiliki minimal 3 karakter.');
    }
  }
  
  set harga(double value) {
    if (value > 0) {
      _harga = value;
    } else {
      print('Gagal memperbarui harga: Harga harus lebih besar dari 0.');
    }
  }
}

void main() {
  // Membuat instance dari class Produk
  var produk1 = Produk('Monitor Samsung', 3500.0);

  print('--- Informasi Produk Awal ---');
  print('Nama Produk: ${produk1.nama}');
  print('Harga Produk: Rp ${produk1.harga.toStringAsFixed(2)}');

  // Menggunakan setter untuk memperbarui nilai
  print('\n--- Memperbarui Produk ---');
  produk1.nama = 'Keyboard Mekanik';
  produk1.harga = 1200.0;

  print('Nama Produk (setelah update): ${produk1.nama}');
  print('Harga Produk (setelah update): Rp ${produk1.harga.toStringAsFixed(2)}');

  // Menguji validasi setter
  print('\n--- Menguji Validasi Setter ---');
  
  // Nama yang tidak valid (kurang dari 3 karakter)
  print('Mencoba mengatur nama menjadi "TV"');
  produk1.nama = 'TV'; // Ini seharusnya tidak mengubah nama
  print('Nama Produk (setelah gagal update nama): ${produk1.nama}');

  // Harga yang tidak valid (kurang dari atau sama dengan 0)
  print('Mencoba mengatur harga menjadi -100.0');
  produk1.harga = -100.0; // Ini seharusnya tidak mengubah harga
  print('Harga Produk (setelah gagal update harga): Rp ${produk1.harga.toStringAsFixed(2)}');

  print('Mencoba mengatur harga menjadi 0.0');
  produk1.harga = 0.0; // Ini seharusnya tidak mengubah harga
  print('Harga Produk (setelah gagal update harga): Rp ${produk1.harga.toStringAsFixed(2)}');

  // Memastikan nilai tidak berubah setelah percobaan yang tidak valid
  print('\n--- Status Akhir Produk ---');
  print('Nama Produk: ${produk1.nama}');
  print('Harga Produk: Rp ${produk1.harga.toStringAsFixed(2)}');
}

Praktik Enkapsulasi Lengkap(Bonus 2)

Buat kelas User dengan:

i.Field private: _username _password, _email (Menggunakan _ agar tidak bisa diakses langsung dari luar class).

ii.Getter untuk semua field digunakan untuk mengambil nilai dari field private.

iii.Setter dengan validasi digunakan untuk mengubah nilai, tetapi dengan aturan:

Username minimal 5 karakter

Password minimal 8 karakter

Email harus mengandung '@'

Buat objek dan coba modifikasi data.

//Nama: Aghista Bunga Kirani
//Nim: 251410027
//Kelas: SI2A


class User {
  // Field private
  String? _username;
  String? _password;
  String? _email;

  // Getter
  String? get username => _username;
  String? get password => _password;
  String? get email => _email;

  // Setter dengan validasi
  set username(String username) {
    if (username.length >= 5) {
      _username = username;
    } else {
      print("Username harus minimal 5 karakter");
    }
  }

  set password(String password) {
    if (password.length >= 8) {
      _password = password;
    } else {
      print("Password harus minimal 8 karakter");
    }
  }

  set email(String email) {
    if (email.contains("@")) {
      _email = email;
    } else {
      print("Email harus mengandung '@'");
    }
  }
}

void main() {
  // Membuat objek
  User user1 = User();

  // Mengisi data menggunakan setter
  user1.username = "aghista";
  user1.password = "password123";
  user1.email = "aghista@gmail.com";

  // Menampilkan data menggunakan getter
  print("Username: ${user1.username}");
  print("Password: ${user1.password}");
  print("Email: ${user1.email}");

  print("\nCoba input yang salah:");

  // Contoh validasi gagal
  user1.username = "abc";
  user1.password = "123";
  user1.email = "aghistagmail.com";
}

//Nama: Aghista Bunga Kirani
//Nim: 251410027
//Kelas: SI2A


class User {
  String _username;
  String _password;

  // Constructor
  User(this._username, this._password);

  // Getter
  String get username => _username;

  // Method login
  bool login(String username, String password) {
    return _username == username && _password == password;
  }

  // Method ganti password
  void gantiPassword(String lama, String baru) {
    if (_password == lama) {
      if (baru.length >= 8) {
        _password = baru;
        print('Password berhasil diubah.');
      } else {
        print('Gagal mengubah password: Password baru harus minimal 8 karakter.');
      }
    } else {
      print('Gagal mengubah password: Password lama tidak sesuai.');
    }
  }
}

void main() {
  // Membuat objek user
  User user1 = User("admin", "12345678");

  // Percobaan login
  if (user1.login("admin", "12345678")) {
    print("Login berhasil");
  } else {
    print("Login gagal");
  }

  // Ganti password
  user1.gantiPassword("12345678", "passwordBaru");

  // Coba login dengan password baru
  if (user1.login("admin", "passwordBaru")) {
    print("Login dengan password baru berhasil");
  } else {
    print("Login gagal");
  }
}

Latihan Kelas Rekening Bank (Bonus 3)

Buat kelas RekeningBank dengan:

Field private: _nomor Rekening, saldo

Method: deposit(), tarik(), cekSaldo()

Validasi: saldo tidak bisa negatif

Buat program sederhana untuk simulasi transaksi.

//Nama: Ratu Klarisa Rahadianti
//Nim: 251410010
//Kelas: SI2A

class RekeningBank {
  String _nomorRekening;
  double _saldo;

  // Constructor
  RekeningBank(this._nomorRekening, this._saldo);

  // Method deposit
  void deposit(double jumlah) {
    if (jumlah > 0) {
      _saldo += jumlah;
      print("Deposit Rp $jumlah berhasil.");
    } else {
      print("Jumlah deposit tidak valid.");
    }
  }

  // Method tarik
  void tarik(double jumlah) {
    if (jumlah > 0 && jumlah <= _saldo) {
      _saldo -= jumlah;
      print("Penarikan Rp $jumlah berhasil.");
    } else {
      print("Penarikan gagal. Saldo tidak cukup atau jumlah tidak valid.");
    }
  }

  // Method cek saldo
  void cekSaldo() {
    print("Nomor Rekening : $_nomorRekening");
    print("Saldo Saat Ini : Rp $_saldo");
  }
}

void main() {
  // Membuat objek rekening
  RekeningBank rekening = RekeningBank("123456789", 1000000);

  print("=== SIMULASI TRANSAKSI BANK ===");

  rekening.cekSaldo();
  print("----------------------");

  // Simulasi deposit
  rekening.deposit(500000);
  rekening.cekSaldo();
  print("----------------------");

  // Simulasi penarikan
  rekening.tarik(300000);
  rekening.cekSaldo();
  print("----------------------");

  // Simulasi penarikan gagal
  rekening.tarik(2000000);
  rekening.cekSaldo();
}
