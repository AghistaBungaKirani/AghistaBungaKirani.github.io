03. Class Sederhana (Motor Listrik)

Pada contoh di bawah ini terdapat class MotorListrik yang memiliki atribut merk dan dayaBaterai. Terdapat pula method akselerasi() dan recharge() sebagai aksi utama dari objek kendaraan tersebut.

DART SOURCE CODE
// Nama   : Aghista Bunga Kirani
// Nim    : 251410027
// Kelas  : SI2A

class MotorListrik {
  String? merk;
  int? dayaBaterai;
  
  void akselerasi() {
    print('Motor $merk sedang melaju kencang...');
  }
  
  void recharge() {
    print('Motor $merk sedang melakukan pengisian daya.');
  }
}

void main() {
  MotorListrik evAghista = MotorListrik();
  
  evAghista.merk = "Gesits G1";
  evAghista.dayaBaterai = 90;
  evAghista.akselerasi();
  evAghista.recharge();
}

04. Object Instantiation

Object Instantiation adalah proses menciptakan wujud nyata dari sebuah rancangan. Pada kode berikut, objek villaAghista dibuat dari class VillaPintar untuk mengontrol keamanan area.

DART SOURCE CODE
// Nama   : Aghista Bunga Kirani
// Nim    : 251410027
// Kelas  : SI2A

class VillaPintar {
  String? lokasi; 
  bool isLocked = true; 
}

void main() {
  VillaPintar villaAghista = VillaPintar();
  
  villaAghista.lokasi = "Uluwatu"; 
  villaAghista.isLocked = false;

  print("Lokasi Villa: ${villaAghista.lokasi}"); 
  print("Status Kunci: ${villaAghista.isLocked ? 'Terkunci' : 'Terbuka'}");
}

07. Encapsulation

Menggunakan class StudioFoto untuk melindungi data jumlah foto (_totalFoto) melalui akses private dan getter.

DART SOURCE CODE
// Nama   : Aghista Bunga Kirani
// Nim    : 251410027
// Kelas  : SI2A

class StudioFoto {
  String namaStudio;
  int _totalFoto; 

  StudioFoto(this.namaStudio, this._totalFoto);

  void ambilFoto(int jumlah) {
    if (jumlah > 0) _totalFoto += jumlah;
  }
  
  int get cekStok => _totalFoto;
}

void main() {
  StudioFoto studioAghista = StudioFoto("Aghista Capture", 100);
  print('Studio: ${studioAghista.namaStudio}'); 
  studioAghista.ambilFoto(25);
  print('Total Koleksi Foto: ${studioAghista.cekStok} file'); 
}

09. Product Class (Cloud)

Class CloudService menerapkan named parameters untuk mengatur detail akun dan kapasitas penyimpanan user secara fleksibel agar data tetap terorganisir.

DART SOURCE CODE
// Nama   : Aghista Bunga Kirani
// Nim    : 251410027
// Kelas  : SI2A

class CloudService {
  final String email, zona; 
  final int kapasitasGb; 

  CloudService({
    required this.email,
    required this.kapasitasGb,
    this.zona = 'Asia-Southeast',
  });
}

void main() {
  CloudService user1 = CloudService(
    email: "aghista@cloud.com",
    kapasitasGb: 100,
    zona: "Global-US",
  );

  CloudService user2 = CloudService(
    email: "bunga@cloud.com",
    kapasitasGb: 15,
  );

  print('User 1: ${user1.email} (Zona: ${user1.zona})'); 
  print('User 2: ${user2.email}'); 
}
