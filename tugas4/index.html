import 'dart:math';

class MathUtils {
  // 1. Faktorial
  static int factorial(int n) {
    if (n < 0) throw Exception('Tidak boleh negatif');
    int hasil = 1;
    for (int i = 1; i <= n; i++) {
      hasil *= i;
    }
    return hasil;
  }

  // 2. Cek bilangan prima
  static bool isPrime(int n) {
    if (n <= 1) return false;
    for (int i = 2; i <= sqrt(n); i++) {
      if (n % i == 0) return false;
    }
    return true;
  }

  // 3. Membulatkan double ke desimal tertentu
  static double roundDouble(double value, int places) {
    double mod = pow(10.0, places).toDouble();
    return (value * mod).round() / mod;
  }

  // 4. Random integer dalam range [min, max]
  static int randomInt(int min, int max) {
    return min + Random().nextInt(max - min + 1);
  }

  // 5. Random double dalam range [min, max]
  static double randomDouble(double min, double max) {
    return min + Random().nextDouble() * (max - min);
  }

  // 6. Rata-rata (mean)
  static double mean(List data) {
    if (data.isEmpty) throw Exception('List kosong');
    return data.reduce((a, b) => a + b) / data.length;
  }

  // 7. Median
  static double median(List data) {
    if (data.isEmpty) throw Exception('List kosong');

    List sorted = List.from(data)..sort();
    int n = sorted.length;

    if (n % 2 == 1) {
      return sorted[n ~/ 2].toDouble();
    } else {
      return (sorted[n ~/ 2 - 1] + sorted[n ~/ 2]) / 2;
    }
  }
}

void main() {
  print('Faktorial 5: ${MathUtils.factorial(5)}');
  print('7 prima? ${MathUtils.isPrime(7)}');
  print('Pembulatan: ${MathUtils.roundDouble(3.14159, 2)}');
  print('Random int: ${MathUtils.randomInt(1, 10)}');
  print('Random double: ${MathUtils.randomDouble(1, 5)}');
  
  List angka = [1, 2, 3, 4, 5];
  print('Mean: ${MathUtils.mean(angka)}');
  print('Median: ${MathUtils.median(angka)}');
}

// Abstract class defining the contract for all shapes
abstract class Bentuk {
  double hitungLuas();
  double hitungKeliling();

  void info() {
    print('Luas: ${hitungLuas()}, Keliling: ${hitungKeliling()}');
  }
}

// Persegi (Square) class implementing Bentuk
class Persegi implements Bentuk {
  double sisi;
  Persegi(this.sisi);

  @override
  double hitungLuas() => sisi * sisi;

  @override
  double hitungKeliling() => 4 * sisi;

  @override
  void info() {
    print('Persegi dengan sisi $sisi memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}.');
  }
}

// Lingkaran (Circle) class implementing Bentuk
class Lingkaran implements Bentuk {
  double radius;
  Lingkaran(this.radius);

  @override
  double hitungLuas() => 3.14 * radius * radius;

  @override
  double hitungKeliling() => 2 * 3.14 * radius;

  @override
  void info() {
    print('Lingkaran dengan radius $radius memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}.');
  }
}

// Segitiga (Triangle) - additional shape to demonstrate extensibility
class Segitiga implements Bentuk {
  double alas;
  double tinggi;
  double sisiMiring;
  
  Segitiga(this.alas, this.tinggi, this.sisiMiring);

  @override
  double hitungLuas() => 0.5 * alas * tinggi;

  @override
  double hitungKeliling() => alas + tinggi + sisiMiring;

  @override
  void info() {
    print('Segitiga dengan alas $alas, tinggi $tinggi, sisi miring $sisiMiring memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}.');
  }
}

void main() {
  print('===== SISTEM BENTUK GEOMETRI =====\n');
  
  // Create instances of different shapes
  Bentuk persegi = Persegi(5.0);
  Bentuk lingkaran = Lingkaran(7.0);
  Bentuk segitiga = Segitiga(3.0, 4.0, 5.0);
  
  // Demonstrate polymorphism by storing shapes in a list
  List daftarBentuk = [persegi, lingkaran, segitiga];
  
  // Process each shape using the abstract interface
  for (var bentuk in daftarBentuk) {
    bentuk.info();
    print('---');
  }
  
  print('\n===== DEMONSTRASI METHOD OVERRIDE =====');
  
  // Direct calls showing polymorphic behavior
  print('Memanggil info() pada Persegi:');
  persegi.info();
  
  print('\nMemanggil info() pada Lingkaran:');
  lingkaran.info();
  
  print('\nMemanggil info() pada Segitiga:');
  segitiga.info();
  
  print('\n===== PERHITUNGAN LANGSUNG =====');
  print('Luas persegi (sisi=5): ${persegi.hitungLuas()}');
  print('Keliling lingkaran (radius=7): ${lingkaran.hitungKeliling()}');
  print('Luas segitiga (alas=3, tinggi=4): ${segitiga.hitungLuas()}');
}

// Abstract class MediaPlayer
abstract class MediaPlayer {
  String name;
  int duration; // in seconds
  
  MediaPlayer(this.name, this.duration);
  
  void displayInfo();
}

// Interface Playable (using abstract class in Dart)
abstract class Playable {
  void play();
  void stop();
}

// Interface Pausable
abstract class Pausable {
  void pause();
  void resume();
}

// AudioPlayer implementation
class AudioPlayer extends MediaPlayer implements Playable, Pausable {
  bool _isPlaying = false;
  bool _isPaused = false;
  int _currentPosition = 0;
  
  AudioPlayer(String name, int duration) : super(name, duration);
  
  @override
  void play() {
    if (_isPaused) {
      print('Resuming audio: $name');
      _isPaused = false;
      _isPlaying = true;
    } else if (!_isPlaying) {
      print('Playing audio: $name (Duration: ${duration} seconds)');
      _isPlaying = true;
      _currentPosition = 0;
    } else {
      print('Audio $name is already playing');
    }
  }
  
  @override
  void stop() {
    if (_isPlaying || _isPaused) {
      print('Stopping audio: $name');
      _isPlaying = false;
      _isPaused = false;
      _currentPosition = 0;
    } else {
      print('Audio $name is not playing');
    }
  }
  
  @override
  void pause() {
    if (_isPlaying && !_isPaused) {
      print('Pausing audio: $name at position $_currentPosition seconds');
      _isPlaying = false;
      _isPaused = true;
    } else if (!_isPlaying) {
      print('Cannot pause $name - audio is not playing');
    } else {
      print('Audio $name is already paused');
    }
  }
  
  @override
  void resume() {
    if (_isPaused) {
      print('Resuming audio: $name from position $_currentPosition seconds');
      _isPlaying = true;
      _isPaused = false;
    } else {
      print('Cannot resume $name - audio is not paused');
    }
  }
  
  @override
  void displayInfo() {
    print('=== Audio Player Info ===');
    print('Name: $name');
    print('Duration: ${duration} seconds');
    print('Status: ${_isPlaying ? 'Playing' : (_isPaused ? 'Paused' : 'Stopped')}');
    print('Current position: $_currentPosition seconds');
    print('========================\n');
  }
}

// Main function to demonstrate the system
void main() {
  print('=== MEDIA PLAYER SYSTEM DEMO ===\n');
  
  // Create instances
  var audioPlayer = AudioPlayer('Summer Vibes.mp3', 180);
  var videoPlayer = VideoPlayer('Nature Documentary.mp4', 360);
  
  // Display initial info
  audioPlayer.displayInfo();
  videoPlayer.displayInfo();
  
  // Test AudioPlayer
  print('--- Testing AudioPlayer ---');
  audioPlayer.play();
  audioPlayer.pause();
  audioPlayer.resume();
  audioPlayer.stop();
  audioPlayer.displayInfo();
  
  print('\n--- Testing VideoPlayer ---');
  videoPlayer.play();
  videoPlayer.setQuality('1080p');
  videoPlayer.pause();
  videoPlayer.resume();
  videoPlayer.stop();
  videoPlayer.displayInfo();
  
  print('\n--- Polymorphism Test ---');
  List players = [audioPlayer, videoPlayer];
  
  for (var player in players) {
    player.displayInfo();
    if (player is Playable) {
      print('This player supports Playable interface');
      (player as Playable).play();
      (player as Playable).stop();
    }
    if (player is Pausable) {
      print('This player supports Pausable interface');
    }
    print('---');
  }
}
